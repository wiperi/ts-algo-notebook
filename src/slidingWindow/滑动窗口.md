def slidingWindow(nums: list[int]):
    # 左闭右开区间，窗口索引包含left，不包含right
	left, right = 0, 0

	while right < len(nums):

		# 检查已经更新的窗口是否合法

			# 缩小窗口

```python
def slidingWindow(nums: list[int]):
    # 左闭右开区间，窗口索引包含left，不包含right
	left, right = 0, 0

	while right < len(nums):

		# 检查即将更新的窗口是否合法
		while 窗口不合法:
			# 缩小窗口
			left = left + 1

		# 更新窗口
		right = right + 1
```
